package com.examly;
import java.util.*;


public class ParticipantManager {

    public static void processNames(ArrayList<String> names,char filterChar){

        HashSet<String> seen = new HashSet<>();
        ArrayList<String> uniqueNames = new ArrayList<>();

        for(String name:names){
            String lower = name.toLowerCase();
            if(!seen.contains(lower)){
                seen.add(lower);
                uniqueNames.add(name);
            }
        }

        System.out.println("Unique Participant Count: "+uniqueNames.size());
        System.out.println("Names starting with '"+filterChar+"':");
        for(String name:uniqueNames){
            if(name.toLowerCase().startsWith(Character.toString(filterChar).toLowerCase())){
                System.out.println(name);
            }
        }
        Collections.sort(uniqueNames,String.CASE_INSENSITIVE_ORDER);
        System.out.println("Sorted Participant List:");
        for(String name: uniqueNames){
            System.out.println(name);
        }

       names.clear();
       names.addAll(uniqueNames);





    }

        boolean updated = false;
        for(int i=0;i<names.size();i++){
            if(names.get(i).equalsIgnoreCase(oldName)){
                names.set(i, newName);
                updated = true;
            }
        }
        if(updated){
            System.out.println("Updated name: " +oldName+" -> "+newName);

        }else{
            System.out.println("Name not found.");

        }
